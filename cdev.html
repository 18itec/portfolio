		id_cell AS identificador, 
		cl.name_cell as celula, 
		(SUBSTRING (ps.fullname FROM 1 FOR 12)||'****') AS lider, 
		SUBSTRING (ps.phone FROM 1 FOR 6)||'****'||SUBSTRING (ps.phone FROM 11) AS celular,
		tlb.name_region
FROM tbl_list_member_cell tlmc 
JOIN tbl_cell cl ON (cl.id = tlmc.id_cell)
JOIN tbl_persons ps ON (ps.id = cl.id_leader )
LEFT JOIN tbl_region tlb ON (tlb.id = cl.id_region )
WHERE cl.id_leader  = 395771
GROUP BY 2,3,4,5,6
ORDER BY cl.name_cell ASC
